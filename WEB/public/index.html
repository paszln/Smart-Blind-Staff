COMPUTER PROGRAMMING (2/2019) PROJECT

1. ความเป็นมาของโครงงาน ปัจจุบันประเทศไทยมีผู้พิการทางสายตาอยุ่จำนวนหนึ่งซึ่งยังไม่ได้รับการอำนวยความสะดวกทางด้านการเดินทาง ทางคณะผู้จัดทำได้เล็งเห็นถึงความสำคัญจุดนี้ จึงได้มีความคิดที่จะทำโครงการนี้ขึ้นมาเพื่ออำนวยความสะดวกแก่ผู้พิการทางสายตา โดยการนำเอาเซนเซอร์อัลตร้าโซนิคเป็นตัวเช็คระยะสิ่งกีดขว้าง โดยมีการประมวลผลและแสดงผลออกมาในรูปแบบของเสียงเพื่อแจ้งเตือนผู้พิการทางสายตา

2. วัตถุประสงค์ของโครงงาน การศึกษาวิจัยมีวัตถุประสงค์ เพื่อทดลองการทางานของไม้เท้าคนตาบอดอัจฉริยะ และ เพื่อหาคุณภาพของไม้เท้าคนตาบอดอัจฉริยะ การวิจัยเป็นการวิจัยเชิงทดลอง (Experimental Research) และทีมงานหวังเป็นอย่างสูงว่าผลการทดลองดใช้จะแสดงให้เห็นถึง การทดลองไม้เท้าคนตาบอดอัจฉริยะ ที่การทำงานของเครื่องมีความเที่ยงตรงสูงและสามารถทำงานได้ตามวัตถุประสงค์ที่กำหนดไว้

3. เครื่องมือที่ใช้จัดทำ – กล่องพลาสติก – ไม้เท้าสำหรับผู้พิการทางสายตา – Ultrasonic Sensor HC-04 – Arduino Uno R3 – Jumperwire – Buzzer – Battery 9v

4. รูปภาพต้นแบบการดีไซน์วงจร

5. Source Code
// defines pins numbers
const int trigPin = 9;
const int echoPin = 10;
const int buzzer = 11;

// defines variables
long duration;
int distance;
int safetyDistance;

void setup() {
pinMode(trigPin, OUTPUT); // Sets the trigPin as an Output
pinMode(echoPin, INPUT); // Sets the echoPin as an Input
pinMode(buzzer, OUTPUT);
Serial.begin(9600); // Starts the serial communication
}

void loop() {
// Clears the trigPin
digitalWrite(trigPin, LOW);
delayMicroseconds(2);

// Sets the trigPin on HIGH state for 10micro seconds
digitalWrite(trigPin, HIGH);
delayMicroseconds(10);
digitalWrite(trigPin, LOW);

// Reads the echoPin, returns the sound wave travel time in microseconds
duration = pulseIn(echoPin, HIGH);
// Calculating the distance
distance= duration*0.034/2;

safetyDistance = distance;
if (safetyDistance <= 50){
digitalWrite(buzzer, HIGH);
}

else{ digitalWrite(buzzer, LOW);
}

// Prints the distance on the Serial Monitor
Serial.print("Distance: ");
Serial.println(distance);
}
